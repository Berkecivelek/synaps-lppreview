# Soru: İki Sum - O(n) çözüm bul
# Aday: Jharet Yılmaz | Backend Engineer

def two_sum(nums: list, target: int) → list:
  seen = {}
  for i, num in enumerate(nums):
    complement = target - num
    if complement in seen:
      return [seen[complement], i]
    seen[num] = i
  return []
